version: "3.9"

services:
  postgres:
    image: postgres:16-alpine
    environment:
      POSTGRES_DB: webradio
      POSTGRES_USER: webradio
      POSTGRES_PASSWORD: changeme123
    volumes:
      - ./postgres-data:/var/lib/postgresql/data
      - ./server/init-db.sql:/docker-entrypoint-initdb.d/01-schema.sql
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U webradio"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped

  backend:
    build:
      context: .
      dockerfile: server/Dockerfile
    environment:
      NODE_ENV: production
      PORT: 4000
      API_PREFIX: /api
      # PostgreSQL connection
      DB_HOST: postgres
      DB_PORT: 5432
      DB_NAME: webradio
      DB_USER: webradio
      DB_PASSWORD: changeme123
      # Legacy exports (optional)
      EXPORT_OUTPUT_DIR: /data/exports
    volumes:
      - ./exports_data:/data
    ports:
      - "4000:4000"
    depends_on:
      postgres:
        condition: service_healthy
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://127.0.0.1:4000/api/health >/dev/null 2>&1 || exit 1"]
      interval: 15s
      timeout: 5s
      retries: 5
    restart: unless-stopped

  frontend:
    build:
      context: .
      dockerfile: Dockerfile
      args:
        VITE_API_BASE_URL: /api
    depends_on:
      backend:
        condition: service_healthy
    ports:
      - "3004:80"
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://127.0.0.1/ >/dev/null 2>&1 || exit 1"]
      interval: 15s
      timeout: 5s
      retries: 5
    restart: unless-stopped

volumes:
  postgres-data:
  exports-data:
